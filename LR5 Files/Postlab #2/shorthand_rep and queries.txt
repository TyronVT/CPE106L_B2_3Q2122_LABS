
SHORTHAND NOTATION 
LEGEND: (*) = PRIMARY KEY

RENTER (*RENTER_NUM, FIRST_NAME, MIDDLE_INITIAL, LAST_NAME, ADDRESS, CITY, STATE, POSTAL_CODE, TELEPHONE, EMAIL_ADDRESS)
RENTER_NUM -> FIRST_NAME, MIDDLE_INITIAL, LAST_NAME, ADDRESS, CITY, STATE, POSTAL_CODE, TELEPHONE, EMAIL_ADDRESS

PROPERTY(*LOC_NUM, LOC_NAME, ADDRESS, CITY, STATE, POSTAL_CODE, UNIT_NUM, SQ_FOOT, BEDROOMS_NUM, BATHROOMS_NUM, MAX_PERSON, BASE_WEEKLY_RATE)
LOC_NUM -> LOC_NAME, ADDRESS, CITY, STATE, POSTAL_CODE, UNIT_NUM, SQ_FOOT, BEDROOMS_NUM, BATHROOMS_NUM, MAX_PERSON, BASE_WEEKLY_RATE

RENTAL_AGREEMENT(*RENTER_NUM, START_DATE, END_DATE, WEEKLY_AMOUNT)
RENTER_NUM -> START_DATE, END_DATE, WEEKLY_AMOUNT

CREATE TABLE RENTER (
    RENTER_NUM VARCHAR(100) PRIMARY KEY, 
    FIRST_NAME VARCHAR(100),
    MIDDLE_INITIAL VARCHAR(100), 
    LAST_NAME VARCHAR(100), 
    ADDRESS VARCHAR(100), 
    CITY VARCHAR(100), 
    STATE VARCHAR(100), 
    POSTAL_CODE VARCHAR(100), 
    TELEPHONE VARCHAR(100), 
    EMAIL_ADDRESS VARCHAR(100)
)

CREATE TABLE PROPERTY (
    LOC_NUM VARCHAR(100) PRIMARY KEY, 
    LOC_NAME VARCHAR(100), 
    ADDRESS VARCHAR(100), 
    CITY VARCHAR(100), 
    STATE VARCHAR(100), 
    POSTAL_CODE VARCHAR(100), 
    UNIT_NUM VARCHAR(100), 
    SQ_FOOT INT, 
    BEDROOMS_NUM INT, 
    BATHROOMS_NUM INT, 
    MAX_PERSON INT,
    BASE_WEEKLY_RATE VARCHAR(100)
)

CREATE TABLE RENTAL_AGREEMENT (
    RENTER_NUM VARCHAR(100) PRIMARY KEY, 
    START_DATE DATE, 
    END_DATE DATE, 
    WEEKLY_AMOUNT MONEY
)
